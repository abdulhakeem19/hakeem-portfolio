import 'package:flutter/material.dart';

// ─────────────────────────────────────────────────
// AfforaColors — single source of truth for color.
// Never write a raw Color(0xFF...) inside a widget.
// Usage: Theme.of(context).extension<AfforaColors>()!
// Or:    context.affora   (via the extension below)
// ─────────────────────────────────────────────────

class AfforaColors extends ThemeExtension<AfforaColors> {
  const AfforaColors({
    required this.bg,
    required this.surface,
    required this.surfaceElevated,
    required this.surfaceFeatured,
    required this.green,
    required this.greenPressed,
    required this.greenTint,
    required this.greenMuted,
    required this.textHero,
    required this.textPrimary,
    required this.textSecondary,
    required this.textMuted,
    required this.border,
    required this.borderEmphasis,
    required this.warning,
    required this.danger,
    required this.info,
  });

  final Color bg;
  final Color surface;
  final Color surfaceElevated;
  final Color surfaceFeatured;

  final Color green;          // financial signal only
  final Color greenPressed;   // active tap state on green CTAs
  final Color greenTint;      // badge fills, uplift bg
  final Color greenMuted;     // nav selected, toggles, chips

  final Color textHero;       // amounts, ring number
  final Color textPrimary;    // card titles, names
  final Color textSecondary;  // body, metadata, sub-labels
  final Color textMuted;      // section headers, captions only

  final Color border;         // rgba green 10%
  final Color borderEmphasis; // rgba green 20%

  final Color warning;
  final Color danger;
  final Color info;

  // ── The one dark instance ──────────────────────
  static const dark = AfforaColors(
    bg:               Color(0xFF07120D),
    surface:          Color(0xFF0D1F16),
    surfaceElevated:  Color(0xFF102019),
    surfaceFeatured:  Color(0xFF162C20),
    green:            Color(0xFF27B07A),
    greenPressed:     Color(0xFF22A06B),
    greenTint:        Color(0x1F27B07A),  // ≈12%
    greenMuted:       Color(0xFF1A4A30),
    textHero:         Color(0xFFF5F7F6),
    textPrimary:      Color(0xFFC8D8CC),
    textSecondary:    Color(0xFF8B9B92),
    textMuted:        Color(0xFF6B8A75),
    border:           Color(0x1A27B07A),  // ≈10%
    borderEmphasis:   Color(0x3327B07A),  // ≈20%
    warning:          Color(0xFFF59E0B),
    danger:           Color(0xFFEF4444),
    info:             Color(0xFF60A5FA),
  );

  @override
  AfforaColors copyWith({
    Color? bg, Color? surface, Color? surfaceElevated,
    Color? surfaceFeatured, Color? green, Color? greenPressed,
    Color? greenTint, Color? greenMuted, Color? textHero,
    Color? textPrimary, Color? textSecondary, Color? textMuted,
    Color? border, Color? borderEmphasis,
    Color? warning, Color? danger, Color? info,
  }) => AfforaColors(
    bg: bg ?? this.bg,
    surface: surface ?? this.surface,
    surfaceElevated: surfaceElevated ?? this.surfaceElevated,
    surfaceFeatured: surfaceFeatured ?? this.surfaceFeatured,
    green: green ?? this.green,
    greenPressed: greenPressed ?? this.greenPressed,
    greenTint: greenTint ?? this.greenTint,
    greenMuted: greenMuted ?? this.greenMuted,
    textHero: textHero ?? this.textHero,
    textPrimary: textPrimary ?? this.textPrimary,
    textSecondary: textSecondary ?? this.textSecondary,
    textMuted: textMuted ?? this.textMuted,
    border: border ?? this.border,
    borderEmphasis: borderEmphasis ?? this.borderEmphasis,
    warning: warning ?? this.warning,
    danger: danger ?? this.danger,
    info: info ?? this.info,
  );

  @override
  AfforaColors lerp(ThemeExtension<AfforaColors>? other, double t) {
    if (other is! AfforaColors) return this;
    return AfforaColors(
      bg:               Color.lerp(bg, other.bg, t)!,
      surface:          Color.lerp(surface, other.surface, t)!,
      surfaceElevated:  Color.lerp(surfaceElevated, other.surfaceElevated, t)!,
      surfaceFeatured:  Color.lerp(surfaceFeatured, other.surfaceFeatured, t)!,
      green:            Color.lerp(green, other.green, t)!,
      greenPressed:     Color.lerp(greenPressed, other.greenPressed, t)!,
      greenTint:        Color.lerp(greenTint, other.greenTint, t)!,
      greenMuted:       Color.lerp(greenMuted, other.greenMuted, t)!,
      textHero:         Color.lerp(textHero, other.textHero, t)!,
      textPrimary:      Color.lerp(textPrimary, other.textPrimary, t)!,
      textSecondary:    Color.lerp(textSecondary, other.textSecondary, t)!,
      textMuted:        Color.lerp(textMuted, other.textMuted, t)!,
      border:           Color.lerp(border, other.border, t)!,
      borderEmphasis:   Color.lerp(borderEmphasis, other.borderEmphasis, t)!,
      warning:          Color.lerp(warning, other.warning, t)!,
      danger:           Color.lerp(danger, other.danger, t)!,
      info:             Color.lerp(info, other.info, t)!,
    );
  }
}

extension AfforaTheme on BuildContext {
  AfforaColors get affora =>
      Theme.of(this).extension<AfforaColors>()!;
}

The critical trap to avoid
Do NOT set colorScheme: ColorScheme.fromSeed(seedColor: Color(0xFF27B07A)) — Material will derive every interactive color from that seed and paint green onto nav bars, switches, checkboxes, and progress indicators. Use greenMuted for the scheme primary instead.
import 'package:flutter/material.dart';
import 'affora_colors.dart';

class AfforaTheme {
  static ThemeData get dark {
    const c = AfforaColors.dark;

    return ThemeData(
      useMaterial3: true,
      brightness: Brightness.dark,
      scaffoldBackgroundColor: c.bg,

      // ── ColorScheme: greenMuted drives all M3 chrome ──
      colorScheme: ColorScheme.dark(
        primary:          c.greenMuted,    // nav, toggles, chips
        onPrimary:        c.textHero,
        primaryContainer: c.surfaceFeatured,
        onPrimaryContainer: c.textPrimary,
        secondary:        c.surface,
        onSecondary:      c.textPrimary,
        surface:          c.surface,
        onSurface:        c.textPrimary,
        surfaceContainerHighest: c.surfaceElevated,
        error:            c.danger,
        onError:          c.textHero,
        outline:          c.border,
        outlineVariant:   c.border,
      ),

      // ── NavigationBar ─────────────────────────────
      navigationBarTheme: NavigationBarThemeData(
        backgroundColor:     c.surface,
        indicatorColor:      c.greenMuted,   // NOT c.green
        iconTheme: WidgetStateProperty.resolveWith((states) {
          if (states.contains(WidgetState.selected)) {
            return IconThemeData(color: c.green, size: 22);
          }
          return IconThemeData(color: c.textMuted, size: 22);
        }),
        labelTextStyle: WidgetStateProperty.resolveWith((states) {
          if (states.contains(WidgetState.selected)) {
            return TextStyle(
              color: c.green, fontSize: 10,
              fontWeight: FontWeight.w600);
          }
          return TextStyle(color: c.textMuted, fontSize: 10);
        }),
      ),

      // ── Cards ─────────────────────────────────────
      cardTheme: CardTheme(
        color: c.surface,
        surfaceTintColor: Colors.transparent,
        shape: RoundedRectangleBorder(
          borderRadius: BorderRadius.circular(12),
          side: BorderSide(color: c.border, width: 0.5),
        ),
        elevation: 0,
      ),

      // ── Typography ────────────────────────────────
      textTheme: TextTheme(
        displayLarge:  TextStyle(color: c.textHero,
                         fontWeight: FontWeight.w600),
        titleLarge:    TextStyle(color: c.textPrimary,
                         fontWeight: FontWeight.w500),
        bodyLarge:     TextStyle(color: c.textPrimary),
        bodyMedium:    TextStyle(color: c.textSecondary),
        bodySmall:     TextStyle(color: c.textMuted),
        labelSmall:    TextStyle(color: c.textMuted,
                         letterSpacing: 0.08),
      ),

      // ── Register the extension ────────────────────
      extensions: const [AfforaColors.dark],
    );
  }
}
Why the nav icon uses c.green but the indicator uses c.greenMuted: The icon is the financial-signal green — it's a positive "you're here" confirmation. The background indicator slab behind it uses greenMuted so it doesn't read as a data signal. The icon is semantic. The slab is chrome.

The one regression to watch for in Phase 1
The existing theme likely uses AppColors.primaryGreen or similar static values directly in widgets. After registering the ThemeExtension, those static references still compile — they just don't go through the theme. Do a project-wide search for Color(0xFF57 and AppColors. before calling Phase 1 done. Every raw reference is a future inconsistency.
